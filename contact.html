    <script>

        function sendEmail() {

            const contactType = document.getElementById('contact-type').value;

            

            if (!contactType) {

                alert('Please select who you are before sending email.');

                return;

            }

            

            let subject = '';

            

            if (contactType === 'parent') {

                subject = 'Contact from a Concerned Parent';

            } else if (contactType === 'rabbi') {

                subject = 'Contact from a Rabbi or Educator';

            } else if (contactType === 'young-man') {

                subject = 'Contact from a Young Man';

            }

            

            // Replace with your actual email address

            window.location.href = 'mailto:heyrabbi@gmail.com?subject=' + encodeURIComponent(subject);

        }
